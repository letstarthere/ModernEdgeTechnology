// Simple calculator in Lingo
store NUM1 as input of "Enter the first number:"
store OP as input of "Enter the operation (+, -, *, /):"
store NUM2 as input of "Enter the second number:"

create function calculate(receive A, B):
    if OP is "+":
        return A + B
    otherwise if OP is "-":
        return A - B
    otherwise if OP is "*":
        return A * B
    otherwise if OP is "/":
        return A / B
    otherwise:
        return 0

store RESULT as calculate(NUM1, NUM2)
print "Result:"
print RESULT
